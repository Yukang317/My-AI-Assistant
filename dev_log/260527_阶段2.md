## 一、代码快照（阶段 2 完成时）

### step21_api.py（SQLite 持久化版，带详细注释）

```python
"""
================================================================================
阶段 1：FastAPI 端点 — 把聊天逻辑变成 HTTP 服务
阶段 2：SQLite 持久化    — 对话历史存到文件，重启不丢失
================================================================================
"""

# ============================================================================
# 第一部分：导入依赖
# ============================================================================
import os
import sqlite3                           # Python 内置，不需要 pip install
from dotenv import load_dotenv
from openai import OpenAI
from fastapi import FastAPI
from fastapi.responses import HTMLResponse
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import uvicorn

# ============================================================================
# 第二部分：加载环境变量 + 创建 OpenAI 客户端
# ============================================================================
load_dotenv()

api_key = os.getenv("DEEPSEEK_API_KEY")
base_url = os.getenv("DEEPSEEK_BASE_URL")
model = "deepseek-chat"
client = OpenAI(api_key=api_key, base_url=base_url)

# ============================================================================
# 第三部分：FastAPI 应用 + CORS + 数据模型
# ============================================================================
app = FastAPI(title="个人助理 API")

# add_middleware 就是给 FastAPI 装一个拦截器——每个请求进来，CORS 中间件先检查来源，
# 然后盖个"放行"的章。没有它，浏览器的 fetch 请求会被直接拦掉。
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],     # "任何来源的纸条我都接受"
    allow_methods=["*"],     # "GET、POST 什么请求都行"
    allow_headers=["*"],     # "任何请求头我都收"
)


class ChatRequest(BaseModel):
    """客户端发来的请求体"""
    question: str          # 用户输入的问题


class ChatResponse(BaseModel):
    """服务端返回的响应体"""
    reply: str             # 助手的回复


# ============================================================================
# 第四部分：SQLite 数据库操作
# ============================================================================

# 数据库文件路径。SQLite 的数据全在一个文件里，没有单独的服务器进程。
# 第一次运行时如果 chat.db 不存在，sqlite3.connect() 会自动创建它。
DB_PATH = "chat.db"

# 会话 ID。目前是单用户，所有对话存在同一个 "default" 会话下。
# 以后再支持多会话（切换会话 = 换一个 session_id 即可）。
SESSION_ID = "default"


def init_db():
    """
    初始化数据库：创建 messages 表（如果已经存在就跳过）。

    表结构：
        id         → 自增主键，每条消息的唯一编号
        session_id → 会话 ID，区分不同对话
        role       → 消息角色：system / user / assistant
        content    → 消息内容
        created_at → 创建时间，自动填当前时间戳
    """
    # sqlite3.connect(DB_PATH) 返回一个"连接对象"，通过它操作数据库
    # 相当于 Python: conn = open("chat.db", mode="database")
    conn = sqlite3.connect(DB_PATH)

    # conn.execute(SQL) 执行一条 SQL 语句
    # CREATE TABLE IF NOT EXISTS = "如果表不存在就创建，存在就啥也不做"（安全，不会覆盖数据）
    # TEXT 是 SQL 里的字符串类型，INTEGER 是整数，TIMESTAMP 是时间戳
    conn.execute("""
        CREATE TABLE IF NOT EXISTS messages (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            session_id TEXT NOT NULL,
            role       TEXT NOT NULL,
            content    TEXT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)

    # commit() = 提交事务，把改动真正写入文件。不调这个等于白干。
    conn.commit()
    # close() = 关闭连接，释放资源。好习惯：用完就关。
    conn.close()


def load_messages(session_id=SESSION_ID):
    """
    从数据库加载指定会话的所有历史消息，按时间顺序排列。

    参数：
        session_id — 会话 ID，默认 "default"

    返回：
        list[dict] — 消息列表，格式跟之前全局 messages 变量一模一样
                     例如：[{"role": "system", "content": "..."}, {"role": "user", "content": "..."}]
    """
    conn = sqlite3.connect(DB_PATH)

    # SELECT role, content → 只查这两列
    # WHERE session_id = ? → 过滤出当前会话的消息（? 是占位符，防 SQL 注入）
    # ORDER BY id → 按 ID 升序排列（即按时间顺序，因为 ID 是自增的）
    rows = conn.execute(
        "SELECT role, content FROM messages WHERE session_id = ? ORDER BY id",
        (session_id,)  # 逗号不能省：(session_id,) 是元组，(session_id) 只是加括号的字符串
    ).fetchall()  # fetchall() 返回所有匹配的行，每行是一个 tuple

    conn.close()

    # 把数据库行转换成 dict 列表，格式跟之前的 messages 变量一致
    # row[0] = role, row[1] = content
    return [{"role": row[0], "content": row[1]} for row in rows]


def save_message(session_id, role, content):
    """
    保存一条消息到数据库。

    参数：
        session_id — 会话 ID
        role       — 消息角色："system" / "user" / "assistant"
        content    — 消息文本内容
    """
    conn = sqlite3.connect(DB_PATH)

    # INSERT INTO messages (...) VALUES (...) → 插入一行数据
    # 三个 ? 占位符按顺序被后面元组里的值替换
    conn.execute(
        "INSERT INTO messages (session_id, role, content) VALUES (?, ?, ?)",
        (session_id, role, content),
    )

    conn.commit()
    conn.close()


# ============================================================================
# 第五部分：核心端点 — POST /chat
# ============================================================================
#  app.post(path, ...) 是 FastAPI 实例上的装饰器方法，意思是：
#  "当有人对我的 path 发 HTTP POST 请求时，调用下面这个函数处理它。"
#  response_model 是 @app.post() 的一个参数名，FastAPI 定义的。
#  自动校验结构，自动转换为 JSON，/docs 页面自动生成文档
@app.post("/chat", response_model=ChatResponse)
def chat(req: ChatRequest):
    """
    接收用户问题，从 SQLite 加载历史 → 调用 DeepSeek → 保存回复到 SQLite。
    """

    # 加载历史消息。调用 load_messages() 获取当前会话的所有历史消息。
    # 如果 messages 为空列表（首次启动），自动补一条 system prompt 并存入数据库。
    messages = load_messages()
    if not messages:
        messages.append({
            "role": "system",
            "content": "你是一个乐于助人的个人助理，用中文回答所有问题。",
        })
        save_message(SESSION_ID, "system", messages[0]["content"])

    # 用户消息：同步写入内存列表 + 数据库
    # 数据库和内存列表要同步——API 调用用内存列表，数据库负责持久化
    messages.append({"role": "user", "content": req.question})
    save_message(SESSION_ID, "user", req.question)

    # 调用 DeepSeek API
    # messages 是一次性把完整对话历史发过去的，模型会从第一条读到当前这条
    # temperature=0.7：0.0 保守 ← 0.7 平衡 → 2.0 放飞
    response = client.chat.completions.create(
        model=model,
        messages=messages,
        temperature=0.7,
    )

    # response.choices[0].message.content：逐层取值路径
    # response → choices 列表 → [0] 第一候选 → message 消息对象 → content 回复文本
    reply_content = response.choices[0].message.content

    # 助手回复：同步写入内存列表 + 数据库
    messages.append({"role": "assistant", "content": reply_content})
    save_message(SESSION_ID, "assistant", reply_content)

    # FastAPI 自动把 ChatResponse 对象序列化成 JSON：{"reply": "..."}
    return ChatResponse(reply=reply_content)


# ============================================================================
# 第六部分：前端页面
# ============================================================================
# 浏览器地址栏输入网址 → 浏览器发 GET 请求（"给我这个页面"）
# HTMLResponse 告诉浏览器"这是网页，请渲染"，不加的话 FastAPI 默认返回 JSON
@app.get("/", response_class=HTMLResponse)
def index():
    with open("static/index.html", "r", encoding="utf-8") as f:
        return f.read()


# ============================================================================
# 第七部分：启动入口
# ============================================================================
if __name__ == "__main__":
    # 服务启动时先初始化数据库（建表），保证后续请求能正常读写
    init_db()
    # host="0.0.0.0" = 任何来源都接（本机、内网、公网）
    # 如果改成 host="127.0.0.1"，就只有 ECS 自己能访问
    uvicorn.run(app, host='0.0.0.0', port=1000)
```

## 二、概念笔记

### 2.1 `isoformat()` — ISO 8601 时间格式

`datetime.datetime.now()` 返回的是 Python 时间**对象**，HTTP/JSON 需要字符串。`.isoformat()` 把对象转成标准格式字符串。

```python
now = datetime.datetime.now()
print(now)                  # 2026-05-27 14:30:00.123456  ← datetime 对象
print(now.isoformat())      # "2026-05-27T14:30:00.123456" ← str
```

**为什么叫 isoformat**：ISO 8601 是国际标准时间格式（`年-月-日T时:分:秒`），全世界的编程语言都认。

**为什么不用 `str(now)`**：各国日期顺序不同（中/美），ISO 格式统一无歧义。

### 2.2 `list_sessions()` — SQL GROUP BY 聚合查询

```python
def list_sessions():
    conn = sqlite3.connect(DB_PATH)
    rows = conn.execute("""
        SELECT session_id,
               (SELECT content FROM messages WHERE session_id = m.session_id
                AND role = 'user' ORDER BY id LIMIT 1) as title,
               COUNT(*) as msg_count,
               MAX(created_at) as updated_at
        FROM messages m
        GROUP BY session_id
        ORDER BY updated_at DESC
    """).fetchall()
    conn.close()
    return [
        SessionInfo(
            session_id=row[0],
            title=row[1] or "新会话",
            message_count=row[2],
            updated_at=row[3],
        )
        for row in rows
    ]
```

**逐行拆解**（按 SQL 实际执行顺序，不是书写顺序）：

| 执行序 | SQL 子句 | 做了什么 |
|--------|---------|---------|
| 1 | `FROM messages m` | 打开表，起别名 `m` |
| 2 | `GROUP BY session_id` | 按 session_id 分组，每组多行浓缩成一行 |
| 3 | `SELECT session_id` | 输出组名 |
| 4 | 子查询 `SELECT content ... LIMIT 1` | 取每组第一条 user 消息作标题 |
| 5 | `COUNT(*)` | 统计每组行数 |
| 6 | `MAX(created_at)` | 取每组最新时间 |
| 7 | `ORDER BY updated_at DESC` | 按最新时间降序，最近聊的排上面 |

**关键理解**：SQL 的**书写顺序**（SELECT → FROM → WHERE → GROUP BY → ORDER BY）和**执行顺序**（FROM → WHERE → GROUP BY → SELECT → ORDER BY）是两套逻辑。

设计哲学是"先说你要什么，再说从哪来"——人先看结果长什么样，再看数据从哪取。但引擎执行时必然先定位数据源。

### 2.3 `sqlite3` 四步操作模式（对着抄即可）

```python
# 查询
conn = sqlite3.connect("chat.db")
rows = conn.execute("SELECT ... WHERE ...", (参数,)).fetchall()
conn.close()

# 写入/删除
conn = sqlite3.connect("chat.db")
conn.execute("INSERT/DELETE ...", (参数,))
conn.commit()   # ← 写入类操作必须 commit
conn.close()
```

**关键点**：`(session_id,)` 逗号不能省——`(x)` 是加括号的普通值，`(x,)` 才是单元素元组。
